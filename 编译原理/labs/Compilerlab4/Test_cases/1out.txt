.intel_syntax noprefix
.extern printf
.global main
.data
format_str:
.asciz "%d\n"
.text


main:
push ebp
mov ebp, esp
sub esp, 8
push 5
pop eax
mov DWORD PTR [ebp - 4], eax
push 3
pop eax
mov DWORD PTR [ebp - 8], eax
mov eax, DWORD PTR [ebp - 4]
push eax
mov eax, DWORD PTR [ebp - 8]
push eax
pop ebx
pop eax
cmp eax, ebx
setle al
movzx eax, al
push eax
pop eax
cmp eax, 0
je .if_end_1
mov eax, DWORD PTR [ebp - 4]
push eax
push offset format_str
call printf
add esp, 8
.if_end_1:
mov eax, DWORD PTR [ebp - 8]
push eax
push offset format_str
call printf
add esp, 8
push 0
pop eax
leave
ret
