int fibonacci(int n) {
    if (n <= 0) {
        return 0;
    }
    if (n == 1) {
        return 1;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

int main() {
    int n=5;
    int i=0;
    while (i<=n){
        int m=n;
        if (i % 2 == 0) {
            println_int(fibonacci(m));
        }
        i=i+1;
    }

    return 0;
}
